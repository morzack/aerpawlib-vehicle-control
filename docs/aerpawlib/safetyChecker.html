 39def serialize_msg(raw_json):
 40    """Compress JSON message using zlib"""
 41    compressed_msg = zlib.compress(raw_json.encode("utf-8"))
 42    return compressed_msg

 45def deserialize_msg(compressed_msg):
 46    """Decompress JSON message using zlib"""
 47    raw_msg = zlib.decompress(compressed_msg).decode("utf-8")
 48    msg = json.loads(raw_msg)
 49    return msg

139    def __init__(self, vehicle_config_filename: str, server_port=14580):
140        # Construct map between function request strings and functions
141        self.REQUEST_FUNCTIONS = {

157        self.include_geofences = [

161        # No go zones to exclude from the geofenced area
162        self.exclude_geofences = [
163            readGeofence(os.path.join(vehicle_config_dir, geofence))
164            for geofence in config["exclude_geofences"]
165        ]
166        self.max_speed = config["max_speed"]
167        self.min_speed = config["min_speed"]
168

170        if self.vehicle_type == "copter":
171            self.max_alt = config["max_alt"]
172            self.min_alt = config["min_alt"]
173
174        # start server BLOCKING
175        self.start_server(server_port)

177    def start_server(self, port):
178        context = zmq.Context()
179        socket = context.socket(zmq.REP)
180        socket.bind(f"tcp://*:{port}")
181
182        print("waiting for messages")
183        while True:
184            raw_msg = socket.recv()
185            message = deserialize_msg(raw_msg)
186            print(f"Received request: {message}")
187            try:
188                function_name = message["request_function"]
189                req_function = self.REQUEST_FUNCTIONS[function_name]
190                params = message["params"]
191                if params is None:
192                    response = req_function(None)
193                else:
194                    response = req_function(*params)
195                socket.send(response)
196            except KeyError as e:
197                socket.send(f"Unimplemented function request <{function_name}>")
198            except Exception as e:
199                socket.send(b"Unknown error!")
200                raise (e)

202    def validate_config(self, config: Dict, vehicle_config_filename: str):
203        """Ensures that the provided config dict contains all necessary parameters.
204        Raises an exception if the config is invalid"""
205        # Check if all required params exist
206        for param in self.REQUIRED_PARAMS:
207            if param not in config:
208                raise Exception(
209                    f"Required parameter {param} not found in {vehicle_config_filename}!"
210                )
211
212        # Ensure the vehicle type is valid
213        if config["vehicle_type"] not in self.VEHICLE_TYPES:
214            raise Exception(
215                f"Vehicle type in {vehicle_config_filename} is invalid! Must be one of {self.VEHICLE_TYPES}"
216            )
217
218        # If the vehicle is a copter ensure copter-specific required params exist
219        if config["vehicle_type"] == "copter":
220            for param in self.REQUIRED_COPTER_PARAMS:
221                if param not in config:
222                    raise Exception(
223                        f"Required copter parameter {param} not found in {vehicle_config_filename}!"

226    def validateWaypointCommand(
227        self, curLoc: Coordinate, nextLoc: Coordinate
228    ) -> Tuple[bool, str]:
229        """
230        Makes sure path from current location to next waypoint stays inside geofence and avoids no-go zones.
231        Returns a tuple (bool, str)
232        (False, <error message>) if the waypoint violates geofence or no-go zone constraints, else (True, "").
233        """
234        print(f"Validating {nextLoc}")
235
236        # Makes sure altitude of next waypoint is within regulations
237        if self.vehicle_type == "copter":
238            if nextLoc.alt < self.min_alt or nextLoc.alt > self.max_alt:
239                return (
240                    False,
241                    "Invalid waypoint. Altitude of %s m is not within restrictions! ABORTING!"
242                    % nextLoc.alt,
243                )
244
245        # Makes sure next waypoint is inside one of the include geofences
246        inside_geofence = False
247        for geofence in self.include_geofences:
248            if inside(nextLoc.lon, nextLoc.lat, geofence):
249                inside_geofence = True
250                break
251        if inside_geofence == False:
252            return (
253                False,
254                "Invalid waypoint. Waypoint (%s,%s) is outside of the geofence. ABORTING!"
255                % (nextLoc.lat, nextLoc.lon),
256            )
257        # Makes sure next waypoint is not in a no-go zone
258        for zone in self.exclude_geofences:
259            if inside(nextLoc.lon, nextLoc.lat, zone):
260                return (
261                    False,
262                    "Invalid waypoint. Waypoint (%s,%s) is inside a no-go zone. ABORTING!"
263                    % (nextLoc.lat, nextLoc.lon),
264                )
265        # Makes sure path between two points does not leave geofence
266        for i in range(len(geofence) - 1):
267            if doIntersect(
268                geofence[i]["lon"],
269                geofence[i]["lat"],
270                geofence[i + 1]["lon"],
271                geofence[i + 1]["lat"],
272                curLoc.lon,
273                curLoc.lat,
274                nextLoc.lon,
275                nextLoc.lat,
276            ):
277                return (
278                    False,
279                    "Invalid waypoint. Path from (%s,%s) to waypoint (%s,%s) leaves geofence. ABORTING!"
280                    % (curLoc.lat, curLoc.lon, nextLoc.lat, nextLoc.lon),
281                )
282
283        # Makes sure path between two points does not enter no-go zone
284        for zone in self.exclude_geofences:
285            for i in range(len(zone) - 1):
286                if doIntersect(
287                    zone[i]["lon"],
288                    zone[i]["lat"],
289                    zone[i + 1]["lon"],
290                    zone[i + 1]["lat"],
291                    curLoc.lon,
292                    curLoc.lat,
293                    nextLoc.lon,
294                    nextLoc.lat,
295                ):
296                    return (
297                        False,
298                        "Invalid waypoint. Path from (%s,%s) to waypoint (%s,%s) enters no-go zone. ABORTING!"
299                        % (curLoc.lat, curLoc.lon, nextLoc.lat, nextLoc.lon),
300                    )
301
302        # Next waypoint location is valid
303        return (True, "")

325    def validateTakeoffCommand(self, takeoffAlt, currentLat, currentLon):
326        """
327        Makes sure the takeoff altitude lies within the vehicle constraints
328        Returns (False, <error message>) if the altitude violates constraints, else (True, "").
329        """

331        if self.vehicle_type == "copter":

339    def validateLandingCommand(self, currentLat, currentLon):
340        """
341        Ensure the copter is attempting to land within 5 meters of the takeoff location
342        Returns (False, <error message>) if the coper is not within 5 meters, else (True, "").
343        """
344        # # approximate radius of earth in km
345        # R = 6373.0
346        # # Calculate distance between takeoff and landing
347        # lat1 = radians(currentLat)
348        # lon1 = radians(currentLon)
349        # lat2 = radians(self.takeoff_location[0])
350        # lon2 = radians(self.takeoff_location[1])
351        # dlon = lon2 - lon1
352        # dlat = lat2 - lat1
353        # a = sin(dlat / 2)**2 + cos(lat1) * cos(lat2) * sin(dlon / 2)**2
354        # c = 2 * atan2(sqrt(a), sqrt(1 - a))
355        # distance = R * c * 1000
356        currentLocation = Coordinate(currentLat, currentLon, alt=0)
357        distance = self.takeoff_location.ground_distance(currentLocation)
358
359        if distance > 5:
360            return (
361                False,
362                "Invalid landing location. Must be within 5 meters of takeoff location. Attempted landing location (%s,%s) is %f meters from takeoff location."
363                % (currentLat, currentLon, distance),
364            )
365        # Landing command is valid
366        return (True, "")

228        self, curLoc: Coordinate, nextLoc: Coordinate
229    ) -> Tuple[bool, str]:
230        """
231        Makes sure path from current location to next waypoint stays inside geofence and avoids no-go zones.
232        Returns a tuple (bool, str)
233        (False, <error message>) if the waypoint violates geofence or no-go zone constraints, else (True, "").
234        """
235        print(f"Validating {nextLoc}")
236
237        # Makes sure altitude of next waypoint is within regulations
238        if self.vehicle_type == "copter":
239            if nextLoc.alt < self.min_alt or nextLoc.alt > self.max_alt:
240                return (
241                    False,
242                    "Invalid waypoint. Altitude of %s m is not within restrictions! ABORTING!"
243                    % nextLoc.alt,
244                )
245
246        # Makes sure next waypoint is inside one of the include geofences
247        inside_geofence = False
248        for geofence in self.include_geofences:
249            if inside(nextLoc.lon, nextLoc.lat, geofence):
250                inside_geofence = True
251                break
252        if inside_geofence == False:
253            return (
254                False,
255                "Invalid waypoint. Waypoint (%s,%s) is outside of the geofence. ABORTING!"
256                % (nextLoc.lat, nextLoc.lon),
257            )
258        # Makes sure next waypoint is not in a no-go zone
259        for zone in self.exclude_geofences:
260            if inside(nextLoc.lon, nextLoc.lat, zone):
261                return (
262                    False,
263                    "Invalid waypoint. Waypoint (%s,%s) is inside a no-go zone. ABORTING!"
264                    % (nextLoc.lat, nextLoc.lon),
265                )
266        # Makes sure path between two points does not leave geofence
267        for i in range(len(geofence) - 1):
268            if doIntersect(
269                geofence[i]["lon"],
270                geofence[i]["lat"],
271                geofence[i + 1]["lon"],
272                geofence[i + 1]["lat"],
273                curLoc.lon,
274                curLoc.lat,
275                nextLoc.lon,
276                nextLoc.lat,
277            ):
278                return (
279                    False,
280                    "Invalid waypoint. Path from (%s,%s) to waypoint (%s,%s) leaves geofence. ABORTING!"
281                    % (curLoc.lat, curLoc.lon, nextLoc.lat, nextLoc.lon),
282                )
283
284        # Makes sure path between two points does not enter no-go zone
285        for zone in self.exclude_geofences:
286            for i in range(len(zone) - 1):

304        return (True, "")
